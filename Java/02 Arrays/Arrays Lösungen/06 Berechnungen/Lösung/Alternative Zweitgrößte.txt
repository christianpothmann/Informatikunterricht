    /*
     * Gibt die zweitgrößte Zahl zurück.
     * Im Sonderfall, dass alle Elemente des Arrays den gleichen Wert haben, 
     * wird eine Mitteilung auf der Konsole gemacht.
     * Algorithmus: Sucht zuerst dax Maximum.
     * Wiederholt dann den Algorithmus, wählt aber nur Zahlen aus, die zugleich kleiner als das Max. sind.
     * Um einen Anfangswert für die zweitgrößte zu finden wird der Sonderfall berücksichtigt, dass
     * das Maximum gleich (ein- oder mehrmals) am Anfang der Liste steht.
     */
    int zweitgroesste()
    {
        int i, j;
        int max1, max2;

        // finde das Maximum
        max1 = liste[0];
        for (i = 1; i < liste.length; i++)
        {
            if (liste[i] > max1)
            {
                max1 = liste[i];
            }
        }

        // springe nach vorn bis eine Zahl kleiner als das Maximum erreicht ist
        j = 0;
        while (liste[j] == max1)
        {
            j++;
        }

        // Sonderfall: alle Elemente der Liste sind gleich
        if (j == liste.length)
        {
            IOTools.println("Es gibt keine zweitgrößte Zahl.");
        }

        // suche die zweitgrößte Zahl
        max2 = liste[j];
        for (i = j+1; i < liste.length; i++)
        {
            if (liste[i] > max2 && liste[i] < max1)
            {
                max2 = liste[i];
            }
        }

        return max2;
    }
